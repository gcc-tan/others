### 分组密码和流密码
这是对明文和密文的处理方式来区分的，密钥K通过比特流生成算法生成比特流，然后与明文的比特流的每个位进行运算这就是流密码。然后如果明文作为一个整体，通过密钥做某种运算，一般输出一个与明文等长的密文分组这就是分组密码

###DES
des的主要工作方式就是代替和置换，des的工作是通过16轮迭代产生的。然或每次的迭代都是以一个F函数为核心，F函数有有两个输入参数Ki，Ri。先说Ki，Ki是通过64位的密钥进行PC1置换（忽略8位），变成56位，然后被分为Ci，Di左右两部分，Ci每次循环左移指定位数，Di每次循环左移指定位数，这个每次左移指定的位数是由一个固定的向量决定的。左移后Ci，Di组成56位的密钥进行PC2置换输出Ki。同时C(i+1),D(i+1)就是左移之后的Ci，Di。而前面的Ri的初始化是由64位的明文进行初始置换IP，然后取左边32位产生Li，右边32位产生Ri，右边是迭代公式Li = R(i-1) , Ri = L(i-1) ^ F{R(i-1),Ki).....具体的流程可以参考《密码编码学与网络安全--原理与实践第五版》Des加密那一节

#define LB32_MASK 0x00000001
#define LB64_MASK 0x0000000000000001
#define L64_MASK 0x00000000ffffffff
#define H64_MASK 0xffffffff00000000

uint64_t des(uint64_t input, uint64_t key, char mode)
{
    int i, j;

    /* 8 bits */
    char row, column;

    /* 28 bits */
    uint32_t C = 0;
    uint32_t D = 0;

    /* 32 bits */
    uint32_t L = 0;
    uint32_t R = 0;
    uint32_t s_output = 0;
    uint32_t f_function_res = 0;
    uint32_t temp = 0;

    /* 48 bits */
    uint64_t sub_key[16] = {0};
    uint64_t s_input = 0;

    /* 56 bits */
    uint64_t permuted_choice_1 = 0;
    uint64_t permuted_choice_2 = 0;

    /* 64 bits */
    uint64_t init_perm_res = 0;
    uint64_t des_result = 0;
    uint64_t pre_output = 0;

    /* IP permutation */
    for (i = 0; i < 64; i++) {
        init_perm_res <<= 1;
        init_perm_res |= (input >> (64-IP[i])) & LB64_MASK;
    }

    L = (uint32_t) (init_perm_res >> 32) & L64_MASK;
    R = (uint32_t) init_perm_res & L64_MASK;

    /* Calculation of the 16 keys */
    for (i = 0; i < 56; i++) {
        permuted_choice_1 <<= 1;
        permuted_choice_1 |= (key >> (64-PC1[i])) & LB64_MASK;
    }

    C = (uint32_t) ((permuted_choice_1 >> 28) & 0x000000000fffffff);
    D = (uint32_t) (permuted_choice_1 & 0x000000000fffffff);

    for (i = 0; i< 16; i++) {
        for (j = 0; j < iteration_shift[i]; j++) {//这个地方把移动化成每次移动一位，这样可能速度有所减慢但是代码简单不少，而且在iteration_shift不大的时候是没什么影响的
            C = (0x0fffffff & (C << 1)) | (0x00000001 & (C >> 27));
            D = (0x0fffffff & (D << 1)) | (0x00000001 & (D >> 27));
        }

        permuted_choice_2 = 0;
        permuted_choice_2 = (((uint64_t) C) << 28) | (uint64_t) D ;

        sub_key[i] = 0;

        for (j = 0; j < 48; j++) {
            sub_key[i] <<= 1;
            sub_key[i] |= (permuted_choice_2 >> (56-PC2[j])) & LB64_MASK;
        }
    }

    for (i = 0; i < 16; i++) {
        /* f(R,k) function */
        s_input = 0;

        for (j = 0; j< 48; j++) {
            s_input <<= 1;
            s_input |= (uint64_t) ((R >> (32-E[j])) & LB32_MASK);
        }

        /* XORing E(Ri) with Ki */
        if (mode == 'd') {//我写这块的时候就是判断是解密直接将sub_key倒置
            // decryption
            s_input = s_input ^ sub_key[15-i];
        } else {
            // encryption
            s_input = s_input ^ sub_key[i];
        }

        /* S-Box Tables */
        for (j = 0; j < 8; j++) {
            //这里对s_input的方式我感觉很值得学习，用一个固定模式的字节0x84 = 100001 通过移动它来取得想要的数据
            row = (char) ((s_input & (0x0000840000000000 >> (6*j))) >> (42-6*j));
            row = (row >> 4) | (row & 0x01);

            column = (char) ((s_input & (0x0000780000000000 >> (6*j))) >> (43-6*j));

            s_output <<= 4;
            s_output |= (uint32_t) (S[j][16*row + column] & 0x0f);
        }

        f_function_res = 0;

        for (j = 0; j < 32; j++) {
            f_function_res <<= 1;
            f_function_res |= (s_output >> (32 - P[j])) & LB32_MASK;
        }

        temp = R;
        R = L ^ f_function_res;
        L = temp;
    }

    pre_output = (((uint64_t) R) << 32) | (uint64_t) L;

    /* PI permutation */
    for (i = 0; i < 64; i++) {
        des_result <<= 1;
        des_result |= (pre_output >> (64-PI[i])) & LB64_MASK;
    }

    return des_result;

}
