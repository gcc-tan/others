#### 函数指针
下面是函数指针及其使用的例子

	#include<stdio.h>
    void show(int param)
    {
    	printf("show:%d\n",param);
    }
    int main(int argc,char **argv)
    {
    	void (* fp1)(int);
        typedef void FP2(int);
        typedef void (* FP3)(int);
        fp1 = show;
        fp1(10);
        fp1 = &show;
        (* fp1)(20);
        
        FP2 *fp2 = show;
        fp2(30);
        fp2 = &show;
        (* fp2)(40);
        
        FP3 fp3 = show;
        fp3(50);
        fp3 = &show;
        (* fp3)(60);
        
        return 0;
    }
